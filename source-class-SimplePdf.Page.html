 16      * Conversion factor from PDF/PostScript points to centimeters
 17      * A point is 1/72 of an inch
 18      * An inch is 2.54 centimeters

 22      */

121      * Draw a line from 1 point to another

123      * @param float $x1 x coordinate (in centimeters) of the point from where to draw the line

137     /**
138      * Write a (multiline / optionally wrapping) text to the page
139      *
140      * @param float $x x coordinate (in centimeters) of the anchor point of the text
141      * @param float $y y coordinate (in centimeters) of the anchor point of the text
142      * @param string $text text to write to the PDF (can contain newlines)
143      * @param string $align either Page::TEXT_ALIGN_LEFT or Page::TEXT_ALIGN_RIGHT, for left or right alignment respectively
144      * @param float $wrapWidth width (in centimeters) to wrap text at, or leave out for no wrapping
145      */
146     public function writeText($x, $y, $text, $align = self::TEXT_ALIGN_LEFT, $wrapWidth = 0)
147     {
148         if ($wrapWidth > 0) {
149             $text = $this->wordWrapText($text, $wrapWidth);
150         }
151 
152         $lineHeight = $this->getLineHeight();
153         foreach (explode(PHP_EOL, $text) as $index => $line) {
154             if (empty($line)) {
155                 continue;
156             }
157 
158             $alignOffset = ($align == self::TEXT_ALIGN_RIGHT ? -$this->getTextWidth($line) : 0);
159             $this->writeLine($x + $alignOffset, $y + $index * $lineHeight, $line);
160         }
161     }

163     /**
164      * Write a single line of text to the page
165      *
166      * @param float $x x coordinate (in centimeters) of the top-left corner where the text should start
167      * @param float $y y coordinate (in centimeters) of the top-left corner where the text should start
168      * @param string $line line to write to the page, should not contain newlines (and will NOT be wrapped)
169      */
170     public function writeLine($x, $y, $line)
171     {
172         $x = $x * self::CM_TO_POINT;
173         $y = ($this->getHeight() - $y) * self::CM_TO_POINT - $this->getFontSize();
174         $this->drawText($line, $x, $y, 'UTF-8');
175     }

177     /**
178      * Word-wrap a text to a certain width, using the current font properties
179      *
180      * @param string $text text to wrap (can already contain some newlines)
181      * @param string $wrapWidth width (in centimeters) to wrap the text to
182      * @return string the same text but with newlines inserted at the specified $wrapWidth
183      */
184     public function wordWrapText($text, $wrapWidth)
185     {
186         $wrappedText = '';
187         foreach (explode(PHP_EOL, $text) as $line) {
188             $words = explode(' ', $line);
189             $currentLine = array_shift($words);
190             while (count($words) > 0) {
191                 $word = array_shift($words);
192                 if ($this->getTextWidth($currentLine . ' ' . $word) > $wrapWidth) {
193                     $wrappedText .= PHP_EOL . $currentLine;
194                     $currentLine = $word;
195                 } else {
196                     $currentLine .= ' ' . $word;
197                 }
198             }
199             $wrappedText .= PHP_EOL . $currentLine;
200         }
201         return ltrim($wrappedText, PHP_EOL);
202     }

214     /**
215      * Calculates how much (horizontal) space a text would use if written to the page, using
216      * the current font properties
217      *
218      * @param string $text text to calculate the width for (should not contain newlines)
219      * @return float width (in centimeters) that the text would use if written to the page
220      */
221     public function getTextWidth($text)
222     {
223         $font = $this->getFont();
224         $fontSize = $this->getFontSize();
225         $text = iconv('UTF-8', 'UTF-16BE', $text);
226         $chars = array();
227         for ($i = 0; $i < strlen($text); $i++) {
228             $chars[] = (ord($text[$i++]) << 8) | ord($text[$i]);
229         }
230         $glyphs = $font->glyphNumbersForCharacters($chars);
231         $widths = $font->widthsForGlyphs($glyphs);
232         return $fontSize * array_sum($widths) / $font->getUnitsPerEm() / self::CM_TO_POINT;
233     }
